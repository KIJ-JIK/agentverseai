# DevFlow AI - Default Demo FastAPI Gateway
from fastapi import FastAPI, Depends, HTTPException
from core.orchestrator import run_agent_pipeline
from core.auth import get_current_user

app = FastAPI(title="DevFlow AI Gateway")

@app.post("/api/sessions")
async def trigger_pipeline(
    prompt: str,
    user = Depends(get_current_user)
):
    # Triggers the autonomous agent pipeline
    if not prompt:
        raise HTTPException(status_code=400, detail="Prompt empty")
    
    session_id = await run_agent_pipeline(
        user_id=user.id,
        feature_prompt=prompt
    )
    return {"status": "success", "session_id": session_id}

# DevFlow AI Platform

DevFlow AI is an autonomous, multi-agent software engineering platform that designs, implements, reviews, and tests features from simple natural language prompts.

## Agent System Architecture

  1. ArchitectAgent — analyzes requests and compiles JSON specification layout.
  2. FrontendDevAgent / BackendDevAgent — writes production React components and FastAPI route handlers.
  3. CodeReviewerAgent — checks code for quality gates, security vulnerabilities, and logic flaws.
  4. QATesterAgent — generates complete pytest suites targeting generated endpoints.
  5. TechWriterAgent — compiles developer documentation and markdown READMEs.
  6. ReleaseManagerAgent — runs final validation gates and merges code to the feature branch.
